1. Largest 1-Bordered Square
Given a 2D grid of 0s and 1s, return the number of elements in the largest square subgrid that has all 1s on its border, or 0 if such a subgrid doesn't exist in the grid.
 
Example 1:
Input: grid = [[1,1,1],[1,0,1],[1,1,1]]
Output: 9
Example 2:
Input: grid = [[1,1,0,0]]
Output: 1
 
Constraints:
  • 1 <= grid.length <= 100
  • 1 <= grid[0].length <= 100
  • grid[i][j] is 0 or 1

now for any index i,j starting from the bottom right of matrices we will check:
let i = 4 and j = 3
len = min(hor[i][j],ver[i][j])     // because a square grid will be formed with all ones on border and if take max (hor,ver) then we may be having 0 on borders similarly for hor or ver  ,that's why we select minimum of hor and ver
len = min(4,4) = 4
now for hor we must be having hor[i-len+1][j]  >= len , otherwise square will not be complete
//ly, ver[i][j-len+1] >= len
hor[4-4+1][3] = 4 >= len
ver[4][3 - 4+1] = 5 >= len
if all conditions are met
then result = len
result = 4
else reduce len by 1 and check again
 area = res*res

CODE:
int largest1BorderedSquare(vector<vector<int>>& grid) {
      if(grid.size() == 0) return 0;
        int res = 0,len = 0;
        int rows = grid.size();
        int cols = grid[0].size();
       int hor[rows][cols],ver[rows][cols];
        for(int i=0;i<rows;i++)
            for(int j=0;j<cols;j++){
                if(j == 0)
                    hor[i][j] = grid[i][j];
                else
                    hor[i][j] = grid[i][j]==0?0:1+hor[i][j-1];
                if(i == 0)
                    ver[i][j] = grid[i][j];
                else
                    ver[i][j] = grid[i][j]==0?0:1+ver[i-1][j];
            }
        for(int i=rows-1;i>=0;i--)
            for(int j=cols-1;j>=0;j--){
                len = min(hor[i][j],ver[i][j]);
                while(len > res){
                    if(hor[i-len+1][j] >= len && ver[i][j-len+1] >= len)
                        {
                        res = len;
                        break;
                        }
                    else
                        len--;
                }
            }
        return res*res;
    }
